/**
* @param {number[][]} times
* @param {number} N
* @param {number} K
* @return {number}
*/
var networkDelayTime = function (times, N, K) {
const costs = Array(N + 1).fill(Infinity);
costs[0] = 0;
costs[K] = 0;
let updated = true;
while (N-- && updated) {
updated = false;
for (const [u, v, w] of times) {
if (costs[u] === Infinity) {
continue;
}
if (costs[u] + w < costs[v]) {
costs[v] = costs[u] + w;
updated = true;
}
}
}
const max = Math.max(...costs);
return max === Infinity ? -1 : max;
};
