/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }

/**
 * @param {ListNode} head
 * @return {boolean}
 */
var hasCycle = function (head) {
    var p = head,
        q = head;
    while (q != null && q.next != null && q.next.next != null) {
        p = p.next;
        q = q.next.next;
        if (p == q) {
            return true;
        }
    }

    return false;
};

/**
 * @param {ListNode} head
 * @return {ListNode}
 */
const detectCycle = function (head, pos) {
    let p = head,
        q = head;
    while (q && q.next && q.next.next) {
        p = p.next;
        q = q.next.next;
        if (p == q) {
            while (head != p) {
                p = p.next;
                head = head.next;
            }
            return p;
        }
    }

    return null;
};
