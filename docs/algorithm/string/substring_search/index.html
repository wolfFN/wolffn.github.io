const genDFA = pat => {
    const keys = Array.from(new Set(Object.values(pat)));
    const dfa = {};
    const M = pat.length;

    for (let i = 0; i < keys.length; i++) {
        if (keys[i] === pat[0]) {
            dfa[keys[i]] = [1];
        } else {
            dfa[keys[i]] = [0];
        }
    }

    for (let X = 0, j = 1; j < M; j++) {
        for (let i = 0; i < keys.length; i++) {
            if (!dfa[keys[i]]) {
                dfa[keys[i]] = [];

            dfa[keys[i]][j] = dfa[keys[i]][X];
        }
        dfa[pat[j]][j] = j + 1;
        X = dfa[pat[j]][X];
    }
    return dfa;
};
